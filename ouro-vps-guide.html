from transformers import AutoTokenizer, AutoModelForCausalLM
import torch
model_path = "./models/ouro-2.6b-thinking"
tokenizer = AutoTokenizer.from_pretrained(model_path)
model = AutoModelForCausalLM.from_pretrained(
    model_path,
    torch_dtype=torch.float16,
    device_map="auto"
)
Devil's Advocate prompt

system_prompt = """You are a Devil's Advocate AI for RCC (Royal Canary Corp).
Your role: challenge assumptions, find flaws in reasoning, stress-test investment decisions.
Be direct, critical, and analytical. Point out risks others miss.
Always respond in Vietnamese unless asked otherwise."""
def ask(question):
    messages = [
        {"role": "system", "content": system_prompt},
        {"role": "user", "content": question}
    ]
    text = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
    inputs = tokenizer([text], return_tensors="pt").to(model.device)
    
    with torch.no_grad():
        output = model.generate(
            **inputs,
            max_new_tokens=512,
            temperature=0.7,
            do_sample=True,
            pad_token_id=tokenizer.eos_token_id
        )
    
    response = tokenizer.decode(output[0][len(inputs.input_ids[0]):], skip_special_tokens=True)
    return response

5. Deploy API Server (FastAPI)

server.py — REST API cho Telegram bot hoặc internal tools

from fastapi import FastAPI
from pydantic import BaseModel
from transformers import AutoTokenizer, AutoModelForCausalLM
import torch, uvicorn
app = FastAPI(title="Ouro Devil's Advocate API")
Load model once at startup

model_path = "./models/ouro-2.6b-thinking"
tokenizer = AutoTokenizer.from_pretrained(model_path)
model = AutoModelForCausalLM.from_pretrained(
    model_path, torch_dtype=torch.float16, device_map="auto"
)
SYSTEM_PROMPT = """Bạn là Devil's Advocate AI của RCC (Royal Canary Corp).
Nhiệm vụ: phản biện giả định, tìm lỗ hổng trong logic, stress-test quyết định đầu tư.
Trả lời trực tiếp, phân tích sắc bén, chỉ ra rủi ro mà người khác bỏ qua.
Trả lời bằng tiếng Việt trừ khi được yêu cầu khác."""
class Query(BaseModel):
    question: str
    context: str = ""  # Optional: paste deal memo, financial model, etc.
@app.post("/ask")
async def ask_devil(query: Query):
    user_msg = query.question
    if query.context:
        user_msg = f"Context:\n{query.context}\n\nCâu hỏi: {query.question}"
    
    messages = [
        {"role": "system", "content": SYSTEM_PROMPT},
        {"role": "user", "content": user_msg}
    ]
    text = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
    inputs = tokenizer([text], return_tensors="pt").to(model.device)
    
    with torch.no_grad():
        output = model.generate(
            **inputs, max_new_tokens=768, temperature=0.7,
            do_sample=True, pad_token_id=tokenizer.eos_token_id
        )
    
    response = tokenizer.decode(output[0][len(inputs.input_ids[0]):], skip_special_tokens=True)
    return {"response": response, "model": "Ouro-2.6B-Thinking"}
@app.get("/health")
async def health():
    return {"status": "ok", "model_loaded": True}
if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8080)

6. Systemd service (auto-restart)

sudo tee /etc/systemd/system/ouro.service > /dev/null <
[Service]
Type=simple
User=$USER
WorkingDirectory=/home/$USER/ouro
Environment="PATH=/home/$USER/ouro/venv/bin"
ExecStart=/home/$USER/ouro/venv/bin/python3 server.py
Restart=always
RestartSec=10
[Install]
WantedBy=multi-user.default
EOF
sudo systemctl daemon-reload
sudo systemctl enable ouro
sudo systemctl start ouro
sudo systemctl status ouro

7. Security (firewall + basic auth)

Chỉ cho phép internal IP access (không expose public)

sudo ufw allow 22/tcp      # SSH
sudo ufw deny 8080/tcp     # Block public access to API
sudo ufw enable
Hoặc nếu muốn Zera gọi từ agent server:

sudo ufw allow from  to any port 8080
